  /**
   * Entrada de objetos tipo Date, salen de ellos DD-Mes-AA en formato string 01-ene-29
   *
   * @param {(Date | undefined | null)} fecha
   * @return {string}
   * @memberof PagareServiceService
   */
  public formatearFecha(fecha: Date | undefined | null): string {
    if (!fecha) return 'no hay fecha';
    const dia = fecha.getDate().toString().padStart(2, '0');
    let mes = (fecha.getMonth() + 1).toString().padStart(2, '0');
    let año = fecha.getFullYear().toString().slice(2);

    switch (mes) {
      case '01':
        mes = 'ene';
        break;
      case '02':
        mes = 'feb';
        break;
      case '03':
        mes = 'mar';
        break;
      case '04':
        mes = 'abr';
        break;
      case '05':
        mes = 'may';
        break;
      case '06':
        mes = 'jun';
        break;
      case '07':
        mes = 'jul';
        break;
      case '08':
        mes = 'ago';
        break;
      case '09':
        mes = 'sep';
        break;
      case '10':
        mes = 'oct';
        break;
      case '11':
        mes = 'nov';
        break;
      case '12':
        mes = 'dic';
        break;
    }

    return `${dia}-${mes}-${año}`;
  }
